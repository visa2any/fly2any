-- Executive dashboard real-time metrics
CREATE VIEW executive_dashboard AS
SELECT 
  -- Revenue Performance
  DATE_TRUNC('month', booking_date) as month,
  SUM(booking_value) as monthly_revenue,
  COUNT(DISTINCT customer_id) as new_customers,
  AVG(booking_value) as avg_transaction_value,
  
  -- Customer Metrics
  COUNT(CASE WHEN booking_date > NOW() - INTERVAL '30 days' THEN 1 END) as monthly_bookings,
  COUNT(DISTINCT CASE WHEN referral_source = 'customer_referral' THEN customer_id END) as referral_customers,
  
  -- Geographic Performance  
  COUNT(CASE WHEN customer_state = 'Connecticut' THEN 1 END) as ct_customers,
  SUM(CASE WHEN customer_state = 'Connecticut' THEN booking_value END) as ct_revenue,
  COUNT(CASE WHEN customer_state = 'Florida' THEN 1 END) as fl_customers,
  SUM(CASE WHEN customer_state = 'Florida' THEN booking_value END) as fl_revenue,
  
  -- Campaign Performance
  SUM(CASE WHEN acquisition_channel = 'email_marketing' THEN booking_value END) as email_revenue,
  SUM(CASE WHEN acquisition_channel = 'phone_outreach' THEN booking_value END) as phone_revenue,
  
  -- Quality Metrics
  AVG(customer_satisfaction_score) as avg_satisfaction,
  COUNT(CASE WHEN customer_satisfaction_score >= 9 THEN 1 END) / COUNT(*) * 100 as promoter_percentage

FROM customer_bookings cb
JOIN customers c ON cb.customer_id = c.id
LEFT JOIN customer_satisfaction cs ON c.id = cs.customer_id
WHERE booking_date >= '2025-09-01'
GROUP BY DATE_TRUNC('month', booking_date)
ORDER BY month;
